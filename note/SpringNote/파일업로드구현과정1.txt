 -저장 경로를 수정하고 싶으면 value의 경로를 수정

public String getFileName() {
	//UUID가 제거되도록 코드작성
	//4d781699-bc2c-4f4e-aa2a-9ebb87ff9d3e_2.jpg => 2.jpg가 되도록
	if(file ==null) {
		return "";
	}
	int index = file.indexOf("_");
	return file.substring(index+1);
}
